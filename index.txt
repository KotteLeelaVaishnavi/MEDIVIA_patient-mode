        // Chat Functions
        function toggleChat() {
            const chatContainer = document.getElementById('chatbotContainer');
            const chatToggle = document.querySelector('.chat-toggle');
            
            if (chatContainer.style.display === 'none' || chatContainer.style.display === '') {
                chatContainer.style.display = 'flex';
                chatContainer.classList.add('active');
                chatToggle.style.display = 'none';
                document.getElementById('chatInput').focus();
            } else {
                chatContainer.style.display = 'none';
                chatContainer.classList.remove('active');
                chatToggle.style.display = 'flex';
            }
        }

        function sendMessage() {
            const input = document.getElementById('chatInput');
            const messagesContainer = document.getElementById('chatMessages');
            const message = input.value.trim();
            
            if (message) {
                // Add user message
                const userMessage = document.createElement('div');
                userMessage.className = 'chat-message user';
                userMessage.textContent = message;
                messagesContainer.appendChild(userMessage);
                
                // Add AI response
                setTimeout(() => {
                    const aiMessage = document.createElement('div');
                    aiMessage.className = 'chat-message ai';
                    aiMessage.textContent = getAIResponse(message);
                    messagesContainer.appendChild(aiMessage);
                    messagesContainer.scrollTop = messagesContainer.scrollHeight;
                    
                    // Check if response requires action
                    handleResponseActions(message);

        function getAIResponse(message) {
            const lowerMessage = message.toLowerCase();
            
            // Symptom tracking
            if (lowerMessage.includes('headache') || lowerMessage.includes('head ache')) {
                return 'I understand you have a headache. Have you taken any pain medication? Would you like me to remind you about your current medications or connect you with Dr. Kabir Singh?';
            }
            if (lowerMessage.includes('dizzy') || lowerMessage.includes('dizziness')) {
                return 'Dizziness can be related to your diabetes or medications. Please check your blood sugar level and consider contacting your doctor if it persists.';
            }
            if (lowerMessage.includes('anxious') || lowerMessage.includes('anxiety') || lowerMessage.includes('stress')) {
                return 'I see you\'re feeling anxious. Remember you\'re taking Sertraline for this. Try some deep breathing exercises. Would you like me to schedule a check-in with your doctor?';
            }
            if (lowerMessage.includes('blood sugar') || lowerMessage.includes('glucose')) {
                return 'Your recent blood sugar readings look good. Remember to take your Metformin twice daily. Would you like me to set a reminder?';
            }
            if (lowerMessage.includes('diarrhea') || lowerMessage.includes('loose motion')) {
                return 'Diarrhea can lead to dehydration, especially with diabetes. Drink plenty of fluids and monitor your blood sugar. If it persists more than 24 hours, contact your doctor.';
            }
            if (lowerMessage.includes('period') || lowerMessage.includes('cramps') || lowerMessage.includes('pms')) {
                return 'For menstrual cramps, try a warm compress and gentle exercise. You can take ibuprofen if needed. Track your symptoms to discuss with your gynecologist.';
            }
            if (lowerMessage.includes('fever') || lowerMessage.includes('temperature')) {
                return 'Please check your temperature. For fever above 101°F (38.3°C) that lasts more than 2 days, contact your doctor. Stay hydrated and rest.';
            }
            if (lowerMessage.includes('cough') || lowerMessage.includes('cold')) {
                return 'For cough/cold symptoms, use your Salbutamol inhaler if needed. Drink warm fluids and rest. If symptoms worsen or last more than 7 days, see your doctor.';
            }
            if (lowerMessage.includes('back pain') || lowerMessage.includes('backache')) {
                return 'For your back pain, continue with physiotherapy exercises. Apply heat for 20 minutes. Avoid heavy lifting. Would you like me to schedule a physio session?';
            }
            
            // Emergency responses
            if (lowerMessage.includes('call doctor') || lowerMessage.includes('contact doctor')) {
                return 'I\'ll help you contact Dr. Kabir Singh. Would you like me to call the emergency line or schedule a regular appointment?';
            }
            if (lowerMessage.includes('emergency') || lowerMessage.includes('urgent')) {
                return 'This seems urgent. I\'m sharing your location and medical summary with the emergency contact: +917416557557. Stay calm, help is on the way.';
            }
            if (lowerMessage.includes('report') || lowerMessage.includes('symptom')) {
                return 'I can help you report symptoms. Please describe what you\'re experiencing in detail (e.g., "I have had a headache for 3 days").';
            }
            
            // Mental wellness
            if (lowerMessage.includes('mental') || lowerMessage.includes('depress') || lowerMessage.includes('sad')) {
                return 'Mental health is important. Would you like to:\n1. Schedule a therapy session\n2. Get mindfulness exercises\n3. Contact your psychiatrist?';
            }
            
            // General health advice
            if (lowerMessage.includes('diet') || lowerMessage.includes('food') || lowerMessage.includes('eat')) {
                return 'For your diabetes, I recommend:\n- High fiber foods (vegetables, whole grains)\n- Lean proteins (fish, chicken)\n- Healthy fats (avocados, nuts)\nWould you like today\'s personalized diet plan?';
            }
            if (lowerMessage.includes('exercise') || lowerMessage.includes('workout') || lowerMessage.includes('activity')) {
                return 'Regular exercise helps manage diabetes and anxiety. Aim for:\n- 30 mins walking daily\n- Yoga or stretching\n- Strength training 2x/week\nWould you like exercise recommendations?';
            }
            
            // Default response
            return 'I\'m here to help with your health concerns. You can ask about symptoms, medications, or request to contact your doctor. How can I assist you today?';
        }

                document.querySelectorAll('.appointment-form').forEach(f => {
                    f.style.display = 'none';
                });
                form.style.display = 'block';
                
                // Set default date to today and time to next hour
                const today = new Date().toISOString().split('T')[0];
                const nextHour = new Date();
                nextHour.setHours(nextHour.getHours() + 1);
                const timeString = nextHour.getHours().toString().padStart(2, '0') + ':00';
                
                form.querySelector('input[type="date"]').value = today;
                form.querySelector('input[type="time"]').value = timeString;
            } else {
                form.style.display = 'none';
            }
        }
